    comparison operator

})
    $in
    1
2
3

    db.students.find({
	"major": {$in: ["design", "chemical"]}	// 大於 1500
})

	{ name: "Sam" },		// 設定條件
	{
		$set: {				// 設定修改後內容
			name: "John",
			major: "Design"
		}
	},
	$currentDate: {			// 顯示最終更新日期
		lastModified: true
	}
